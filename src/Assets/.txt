import React from "react";
import { MapContainer, TileLayer, Marker, Popup } from "react-leaflet";
import { Icon, DivIcon, point, LatLngTuple } from "leaflet";
import "leaflet/dist/leaflet.css";
import "./Maps.css";
import "../Assets/pin.png";
import MarkerClusterGroup from "react-leaflet-cluster";

interface MarkerData {
    geocode: LatLngTuple;
    popUpL: string;
}

const Maps: React.FC = () => {
    const markers: MarkerData[] = [
        {
            geocode: [48.86, 2.35222],
            popUpL: "Hello, I am pop up 1"
        },
        {
            geocode: [48.86, 2.35222],
            popUpL: "Hello, I am pop up 2"
        },
        {
            geocode: [48.86, 2.35222],
            popUpL: "Hello, I am pop up 3"
        }
    ];

    const customIcon = new Icon({
        iconUrl: "../Assets/pin.png", // Make sure the path is correct
        iconSize: [30, 38] as [number, number]
    });

    const createCustomClusterIcon = (cluster: any) => {
        return new DivIcon({
            html: `<div class="cluster-icon">${cluster.getChildCount()}</div>`,
            className: "custom-marker-cluster",
            iconSize: point(33, 33, true)
        });
    };

    return (
        <MapContainer center={{ lat: 48.8566, lng: 2.3522 }} zoom={12} style={{ height: "100vh", width: "100%" }}>
            <TileLayer url='https://tile.openstreetmap.org/{z}/{x}/{y}.png' />
            <MarkerClusterGroup chunkedLoading iconCreateFunction={createCustomClusterIcon}>
                {markers.map((marker, index) => (
                    <Marker key={index} position={marker.geocode} icon={customIcon}>
                        <Popup>
                            <h2>{marker.popUpL}</h2>
                        </Popup>
                    </Marker>
                ))}
            </MarkerClusterGroup>
        </MapContainer>
    );
};

export default Maps;
